# Навигация по системе

pwd - показывает где я (/home/vxuser)
w - показывает кто сейчас работает в системе и что делает
whoami - показывает текущего пользователя
which - показывает где находится программа (Which python)- /usr/bin/python
# Информация о системе

nproc - сколько процессорных ядер
free -h - показывает оперативную память
Поля: 
- total - всего памяти
- used - используется
- free - свободно
# Мониторинг процессоров

top - показывает процессы и нагрузку системы

    

🔹 1. Базовые команды
pwd        # где ты сейчас
whoami     # текущий пользователь 
w          # кто в системе и что делает
which ls   # где лежит команда
nproc      # количество CPU
  

🔹 2. Память и нагрузка
free -h    # оперативка
top        # процессы (в реальном времени)
htop       # удобный top (если установлен)

  

🔹 3. Процессы
ps -edf          # список процессов
kill 1234        # мягко завершить (SIGTERM)
kill -9 1234     # жёстко убить (SIGKILL)
Разница:
- kill -15 (по умолчанию) — аккуратно
- kill -9 — принудительно (если зависло)  

🔹 4. Файлы и папки
mkdir test       # создать папку
cd test          # перейти
touch file.txt   # создать файл
ls -lah          # список файлов
ll               # алиас ls -l
  

🔹 5. Копирование / удаление
cp file1 file2        # копия файла
cp -r dir1 dir2       # копия папки
mv file newname       # переименовать / переместить
rm file               # удалить файл
rm -r dir             # удалить папку

  

🔹 6. Просмотр файлов
cat file.txt
less file.txt
head -n 10 file.txt
tail -n 10 file.txt
